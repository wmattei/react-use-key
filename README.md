## Install

```bash
npm install --save react-use-key
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-use-key'
import 'react-use-key/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
